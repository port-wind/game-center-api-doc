Query Game Statistics Interface

POST /game-proxy/game/order/stat

Request Body

{
  "jsonrpc": "2.0",
  "id": "req-001",
  "method": "game.order.stat",
  "params": {
    "a0": 19191,// Root Node ID，Required
    "userIds":[199817,12123],// User IDArray，Required
    "currency":"RNB",// Currency，Required
    "isSettle":0,// YesNoAlreadySettled 0 No 1Yes Optional
    "state":[],// Status 0，1，2，3，4，5，7   Optional
    "gameInfo":{
         "category":"0",// Game Categoryid Required
         "providerId":"1",// Game Providerid Required
         "gameId":"101891"// Game ID Optional
     },
    "groupId":"DEF",// groupId DefaultDEF Optional
    "beginTime":"YYYY-MM-DD HH:mm:ss",// Start Timestamp Required
    "endTime":"YYYY-MM-DD HH:mm:ss",// End Timestamp Required
    "timezone":0,// Singapore Time 0 ，Eastern Time 1 Required
    "page":1,// Page Number Required
    "size":100,// Item Count Required
    "hasStat":0// YesNoNeed Total 0 No 1 Yes Required
  }
}

{
  "jsonrpc": "2.0",
  "id": "req-001",
  "result": {
      "total":100,// Total Count
      "totalBetScore":0,// Bet AmountTotal

      "totalValidScore":0,// Total Valid Bet Amount
      "list":[
          {
               "gameId":"1",// Game ID
               "groupId":"1",// Roomid
               "currency":"RMB",// Currency
               "betScore":100.0,// Bet Amount
               "settleScore":100.0,// Payout Amount
               "validScore":100.0// Valid Bet Amount
               "userId":100211// User ID
          }
     ]
  
   },
  "error": null

Get Game URL

POST /game-proxy/game/url

gameInfo Parameter Description
"gameInfo":{
         "category":"0",//Game Category ID, Required
         "providerId":"1",//Game Provider ID, Required
         "gameId":"101891"//Game ID，Must provide game type（BGFishing (Game)）
     },

Request Headers

  • Content-Type: application/json
  • token: 70d79fa3-e2f4-46ae-9b28-bbee4cd795b3 See generation method: https://pb-api-doc.pwtk.cc/project/253/wiki

Request Body

{
  "jsonrpc": "2.0",
  "id": "req-001",
  "method": "game.url",
  "params": {
    "token":"sdfdsfdsfsd",// User Token, Required
    "device":"PC",// Terminal, Required
    "language":"CN",// Language, Required
    "currency":"RNB",// Currency, Required
    "groupId":"1000",// Group ID，DefaultDEF，Optional，When isGroupTrace=1 is required
    "isGroupTrace":0,// 0=No Tracking（Default），1=Tracking（Must Fill groupId）
    "isDemo":0,// Is Demo 0No1Yes, Required
    "gameInfo":{
         "category":"0",// Game Category ID, Required
         "providerId":"1",// Game Provider ID, Required
         "gameId":"101891"// Game ID，Optional
     },
     "returnUrl":""// Return URL，Optional
  }
}

Response Body

{
  "jsonrpc": "2.0",
  "id": "req-001",
  "result": {
    "url": "https:// www.ddf.com",//Game Entry URL
    "config": {}
  },
  "error": null
}

Reference
{

"jsonrpc": "2.0",
  "id": "req-001",
  "method": "game.order.collect",
  "params": {
  }
}



Result is paginated data

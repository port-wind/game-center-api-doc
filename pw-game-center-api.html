User Score Change

POST /game-proxy/user/score/change

Request Headers

  • token: 70d79fa3-e2f4-46ae-9b28-bbee4cd795b3

Request Body

{
   "jsonrpc":"2.0",
   "id":"req-001",
   "method":"user.score.change",
   "params":{
       "a0":1008171,// Root Node ID
       "userId":1019818,// User ID
       "category":"0",// Game Category ID, Required
        "providerId":"1",// Game Provider ID, Required
       "gameId":"181818",// Game ID
       "score":999.00,// Changed Score (Greater than 0)
       "transactionId":23424324324",// Order/Transaction ID
       "opType":"debit",// Change Type, debit/credit/return(cancel-refund)
       "currency":"RMB"// Currency
   }
}

Response Body

{
   "jsonrpc":"2.0",
   "id":"req-001",
   "method":"user.score.change",
   "result":{
        "a0":100841,// Root Node
        "userId":100821,// User ID
        "currency":"RMB",// Currency
        "score":"100.00"// Balance After Change
   },
   "error":null
}
